from pyscript import document def calculate_percentage(): try: j = float(document.getElementById("total-number").value) l = float(document.getElementById("obtained-number").value) percentage = (l / j) * 100 document.getElementById("result").innerText = f"The percentage of given numbers is {percentage:.2f}%" except ZeroDivisionError: document.getElementById("result").innerText = "Cannot divide by zero!" except ValueError: document.getElementById("result").innerText = "Invalid input" document.getElementById("calculate-button").addEventListener("click", calculate_percentage)
